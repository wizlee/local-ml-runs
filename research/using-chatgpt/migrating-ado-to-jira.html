        const issueData = {
            fields: {
                project: {
                    key: jiraProjectKey
                },
                issuetype: {
                    name: jiraIssueType
                },
                summary: workItem.fields['System.Title'],

                priority: {

                },

                },
                reporter: {
                    name: jiraUsername
                },
                created: new Date(workItem.fields['System.CreatedDate']),
                status: {
                    name: workItem.fields['System.State']
                },
                resolution: {
                    name: workItem.fields['System.Reason'] || 'Done'
                },
                labels: workItem.fields['System.Tags']
            }
        };

        await axios.post(`${jiraUrl}/issue`, issueData, {

                username: jiraUsername,

            }

        })
        .catch((error) => {
            console.error(`Error creating JIRA issue for Azure DevOps work item ${workItem.id}: ${error.message}`);
        });
    }
})
.catch((error) => {
    console.error(`Error retrieving Azure DevOps work items: ${error.message}`);
});
